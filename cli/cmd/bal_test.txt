package main

import (
	"bytes"
	"strings"
	"sync"
	"testing"

	"github.com/spf13/cobra"
)

// testMu protects rootCmd during tests since it's global state
var testMu sync.Mutex

// createTestRootCmd creates a fresh root command for testing.
func createTestRootCmd() *cobra.Command {
	cmd := &cobra.Command{
		Use:           "bal",
		Short:         "The build system and package manager of Ballerina",
		Long:          "The build system and package manager of Ballerina",
		SilenceUsage:  true,
		SilenceErrors: true,
		Version:       Version,
	}
	cmd.SetVersionTemplate("Ballerina Version: {{.Version}}\n")
	cmd.AddCommand(createNewCmd())
	cmd.AddCommand(runCmd)
	cmd.AddCommand(versionCmd)
	return cmd
}

// =============================================================================
// Root Command Tests
// =============================================================================

// TestRootCmd_Properties tests the root command has correct properties.
func TestRootCmd_Properties(t *testing.T) {
	t.Parallel()

	if rootCmd.Use != "bal" {
		t.Errorf("expected Use to be 'bal', got %q", rootCmd.Use)
	}

	if rootCmd.Short == "" {
		t.Error("expected Short description to be set")
	}

	if rootCmd.Long == "" {
		t.Error("expected Long description to be set")
	}

	if !rootCmd.SilenceUsage {
		t.Error("expected SilenceUsage to be true")
	}

	if !rootCmd.SilenceErrors {
		t.Error("expected SilenceErrors to be true")
	}
}

// TestRootCmd_Help tests that help output contains expected information.
func TestRootCmd_Help(t *testing.T) {
	t.Parallel()

	cmd := createTestRootCmd()
	var outBuf bytes.Buffer
	cmd.SetOut(&outBuf)
	cmd.SetArgs([]string{"--help"})

	err := cmd.Execute()
	if err != nil {
		t.Fatalf("help command failed: %v", err)
	}

	output := outBuf.String()

	// Verify help contains command name
	if !strings.Contains(output, "bal") {
		t.Errorf("help output should contain 'bal', got: %s", output)
	}

	// Verify help mentions available commands
	if !strings.Contains(output, "Available Commands") {
		t.Errorf("help output should contain 'Available Commands', got: %s", output)
	}

	// Verify subcommands are listed
	if !strings.Contains(output, "new") {
		t.Errorf("help output should list 'new' command, got: %s", output)
	}
}

// =============================================================================
// Subcommand Registration Tests
// =============================================================================

// TestRootCmd_HasSubcommands tests that expected subcommands are registered.
func TestRootCmd_HasSubcommands(t *testing.T) {
	t.Parallel()

	cmd := createTestRootCmd()
	commands := cmd.Commands()

	expectedCommands := map[string]bool{
		"new":     false,
		"run":     false,
		"version": false,
	}

	for _, c := range commands {
		// Use c.Name() which returns just the command name without args
		if _, ok := expectedCommands[c.Name()]; ok {
			expectedCommands[c.Name()] = true
		}
	}

	for name, found := range expectedCommands {
		if !found {
			t.Errorf("expected subcommand %q to be registered", name)
		}
	}
}

// TestRootCmd_UnknownCommand tests error handling for unknown commands.
func TestRootCmd_UnknownCommand(t *testing.T) {
	t.Parallel()

	cmd := createTestRootCmd()
	var errBuf bytes.Buffer
	cmd.SetErr(&errBuf)
	cmd.SetArgs([]string{"unknowncommand"})

	err := cmd.Execute()
	if err == nil {
		t.Fatal("expected error for unknown command")
	}

	// Cobra returns an error for unknown commands
	if !strings.Contains(err.Error(), "unknown command") {
		t.Errorf("expected 'unknown command' error, got: %v", err)
	}
}

// TestRootCmd_NoArgs tests root command with no arguments shows help.
func TestRootCmd_NoArgs(t *testing.T) {
	t.Parallel()

	cmd := createTestRootCmd()
	var outBuf bytes.Buffer
	cmd.SetOut(&outBuf)
	cmd.SetArgs([]string{})

	err := cmd.Execute()
	if err != nil {
		t.Fatalf("root command with no args failed: %v", err)

		t.Errorf("expected help output, got: %s", output)
	}
}
